#include <stdio.h>
#include <string.h>
#include <ctype.h>

#define MAX_PROD 10
#define MAX_TERM 10

char productions[MAX_PROD][MAX_TERM];
int n_prod;

int is_terminal(char c) {
    return islower(c) || c == '$';
}

int is_nonterminal(char c) {
    return isupper(c);
}

void first(char c) {
    int i;
    for (i = 0; i < n_prod; i++) {
        if (productions[i][0] == c) {
            if (is_terminal(productions[i][2])) {
                printf("%c ", productions[i][2]);
            } else if (is_nonterminal(productions[i][2])) {
                first(productions[i][2]);
            }
        }
    }
}

void follow(char c) {
    int i, j;
    if (productions[0][0] == c) {
        printf("$ ");
    }
    for (i = 0; i < n_prod; i++) {
        for (j = 2; j < strlen(productions[i]); j++) {
            if (productions[i][j] == c) {
                if (j == strlen(productions[i])-1) {
                    follow(productions[i][0]);
                } else if (is_terminal(productions[i][j+1])) {
                    printf("%c ", productions[i][j+1]);
                } else if (is_nonterminal(productions[i][j+1])) {
                    first(productions[i][j+1]);
                }
            }
        }
    }
}

int main() {
    int i;
    printf("Enter number of productions: ");
    scanf("%d", &n_prod);
    printf("Enter the productions:\n");
    for (i = 0; i < n_prod; i++) {
        scanf("%s", productions[i]);
    }
    printf("First sets:\n");
    for (i = 0; i < n_prod; i++) {
        printf("First(%c) = { ", productions[i][0]);
        first(productions[i][0]);
        printf("}\n");
    }
    printf("\nFollow sets:\n");
    for (i = 0; i < n_prod; i++) {
        printf("Follow(%c) = { ", productions[i][0]);
        follow(productions[i][0]);
        printf("}\n");
    }
    return 0;
}
